# Prompt the user to enter their age and purchase amount
age = int(input("Enter your age: "))
purchase_amount = float(input("Enter your purchase amount: "))

if age < 12 or age > 65:
    discount = 0.15
elif 12 <= age <= 65 and purchase_amount > 100:      # Discount for adults with a purchase > $100
    discount = 0.10
else:
    discount = 0

discount_amount = purchase_amount * discount
final_price = purchase_amount - discount_amount

# Display the discount amount and the final price
print(f"Discount: ${discount_amount} and Final price: ${final_price}")

count = int(input("How many numbers do you want to enter?: "))
total = 0

for i in range(count):
    number = float(input(f"Enter number {i + 1}: "))
    total += number

average = total / count 

print(f"Sum of the numbers: {total}")
print(f"Average of the numbers: {average}")

Question 8:

import random

number = random.randint(1, 100)
guesses = 0

while True:
    guess = int(input("Guess a number between 1 and 100: "))
    guesses += 1

    if guess < number:
        print("Too low.")
    elif guess > number:
        print("Too high.")
    else:
        print(f"Correct! You guessed it in {guesses} tries.")
        break
